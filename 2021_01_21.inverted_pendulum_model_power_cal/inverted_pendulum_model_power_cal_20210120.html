import math
import numpy
import sympy

RPM2RPS = 2*math.pi/60
RPS2DEG = 180/math.pi
INCH2M = 0.0254
MPS2KMH = 3.6

def cal_Kt_from_Kv(Kv):
    RPM2RPS = 2*math.pi/60
    return 1/(Kv*RPM2RPS)
